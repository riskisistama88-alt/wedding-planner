Integrasi Google Apps Script (GAS)

Langkah menghubungkan dasbor ini dengan Google Sheets & Drive milik Anda

Cara Memasang Backend Serverless:

  1. Buat Spreadsheet baru di Google Sheets, lalu klik menu Extensions > Apps Script.
  2. Hapus semua kode bawaan di dalam berkas editor, salin seluruh kode di bawah ini lalu tempel.
  3. Sesuaikan ID kelompok pernikahan atau ID folder di GAS jika diperlukan, lalu klik ikon simpan.
  4. Klik Deploy > New Deployment, pilih jenis Web App.
  5. Atur Execute as: Me (email Anda) dan Who has access: Anyone.
  6. Salin URL Web App yang dihasilkan, lalu masukkan pada kolom pengaturan Web App URL di dasbor di bawah ini untuk menghubungkan database secara permanen!

/* CODE.GS — BACKEND SERVERLESS INTERACTIVE WEDDING PLANNER */
const SPREADSHEET_ID = "MASUKKAN_ID_SPREADSHEET_ANDA_DISINI";

function doGet(e) {
  const action = e.parameter.action;
  const weddingId = e.parameter.wedding_id;
  
  if (action === "getVendorCatalog") {
    return handleGetVendorCatalog(weddingId);
  } else if (action === "getBudgetSummary") {
    return handleGetBudgetSummary(weddingId);
  }
  
  return ContentService.createTextOutput(JSON.stringify({success: false, message: "Invalid GET Action"}))
                       .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const payload = JSON.parse(e.postData.contents);
  const action = payload.action;
  
  if (action === "login") {
    return handleLogin(payload.email, payload.token);
  } else if (action === "addVendor") {
    return handleAddVendor(payload);
  } else if (action === "updateVendorStatus") {
    return handleUpdateVendorStatus(payload);
  } else if (action === "deleteVendor") {
    return handleDeleteVendor(payload);
  }
  
  return ContentService.createTextOutput(JSON.stringify({success: false, message: "Invalid POST Action"}))
                       .setMimeType(ContentService.MimeType.JSON);
}
